
# Install python3 virtualenv in /opt so we can pull in some packages;
#
# In particular, we want to install pylint so we can
# lint our python code in emacs;

cd /opt
virtualenv -p /usr/bin/python3 python3
python3/bin/pip install ipython
python3/bin/pip install mypy
python3/bin/pip install pylint
python3/bin/pip install pytest

# web stuff + django dev
python3/bin/pip install requests
python3/bin/pip install flask
python3/bin/pip install django
python3/bin/pip install djangorestframework
python3/bin/pip install psycopg2

# misc
python3/bin/pip install lxml
python3/bin/pip install pygments
